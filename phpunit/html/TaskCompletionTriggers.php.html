6 * Class for handling task completion triggers.
7 *
8 * Tasks that have smart or automatic completion are managed here.
9 * Each task type has:
10 * - Task path constants (TASK_PATHS) for single source of truth

12 * - Handler methods that respond to WordPress/plugin events

15 * Architecture:

19 * - Validators reuse the same logic as handlers for consistency
20 *
21 * File Organization:
22 * # Constructor & Setup

25 * # Blog Tasks (Content Creation)

162     */
163    private static function has_enabled_payment_gateways(): bool {
164        // Check if WooCommerce is active and loaded
165        if ( ! function_exists( 'WC' ) || ! WC() ) {
166            return false;

169        // Check if payment gateways are available
170        // This filters for enabled AND properly configured payment methods
171        // So a user must enable and configure the payment method to complete the task
172        $payment_gateways = WC()->payment_gateways();
173        if ( ! $payment_gateways ) {
174            return false;
175        }
176
177        // Get available payment gateways
178        $available_gateways = $payment_gateways->get_available_payment_gateways();
179
180        // If any gateways are available, payment setup is complete
181        return ! empty( $available_gateways );
182    }
