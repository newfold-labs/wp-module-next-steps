183            wp_safe_redirect( admin_url() );
184            exit;
185        }

188        $partner_plugins = self::get_partner_plugins();
189        if ( ! array_key_exists( $plugin_slug, $partner_plugins ) ) {

199        // Check plugin status

201
202        // Determine redirect URL based on plugin status
203        $final_redirect_url = $plugin_active ? $final_redirect_url : $final_fallback_url;
204
205        // Perform redirect
206        wp_safe_redirect( $final_redirect_url );
207        exit;
208    }

226        }

242            $template_url = str_replace( '{siteUrl}', untrailingslashit( get_site_url() ), $template_url );

244        } else {
245            // Use fallback URL (defaults to site editor)

247        }
248        exit;
249    }
