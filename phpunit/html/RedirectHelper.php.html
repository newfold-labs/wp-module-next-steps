144        if (

183        if ( empty( $plugin_slug ) ) {
184            // Redirect to admin dashboard if no plugin specified
185            wp_safe_redirect( admin_url() );
186            exit;
187        }
188
189        // Get partner plugin defaults
190        $partner_plugins = self::get_partner_plugins();
191        if ( ! array_key_exists( $plugin_slug, $partner_plugins ) ) {
192            // Redirect to admin dashboard if invalid plugin specified
193            wp_safe_redirect( admin_url() );

195        }

198        $final_redirect_url = ! empty( $redirect_url ) ? $redirect_url : admin_url( $partner_plugins[ $plugin_slug ]['redirect_url'] );
199        $final_fallback_url = ! empty( $fallback_url ) ? $fallback_url : admin_url( $partner_plugins[ $plugin_slug ]['fallback_url'] );
200
201        // Check plugin status
202        $plugin_active = self::check_plugin( $plugin_slug );
203
204        // Determine redirect URL based on plugin status
205        $final_redirect_url = $plugin_active ? $final_redirect_url : $final_fallback_url;
206
207        // Perform redirect
208        wp_safe_redirect( $final_redirect_url );
209        exit;

213     * Handle template-based redirects with real-time URL generation
214     *

218        // Sanitize and validate parameters

220        $template_param = isset( $_GET['template'] ) ? sanitize_text_field( wp_unslash( $_GET['template'] ) ) : 'index';

223        // Set appropriate fallback URL based on theme type if none provided
224        if ( empty( $fallback_url ) ) {

227                : admin_url( 'customize.php' );  // Classic themes: use customizer
228        }
229
230        // Generate the appropriate template URL based on parameters
231        $template_url = null;
232
233        if ( ! empty( $area_param ) ) {
234            // Template part redirect (e.g., header, footer)
235            $template_url = self::get_template_part_editor_url( $area_param, $template_param );
236        } else {
237            // Template redirect (e.g., home, index)
238            $template_url = self::get_template_editor_url( $template_param );
239        }
240
241        // Use template URL if available, otherwise use fallback
242        if ( $template_url ) {
243            // Replace {siteUrl} placeholder with actual site URL
244            $template_url = str_replace( '{siteUrl}', untrailingslashit( get_site_url() ), $template_url );
245            wp_safe_redirect( $template_url );
246        } else {
247            // Use fallback URL (defaults to site editor)
248            wp_safe_redirect( $fallback_url );
249        }
250        exit;
251    }
